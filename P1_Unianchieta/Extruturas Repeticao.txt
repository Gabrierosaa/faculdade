#Laço while
Repete enquanto a condição for verdadeira.

contador = 1
while contador <= 5:
    print(contador)
    contador += 1

#Laço for com range()
Itera sobre uma sequência de valores.

for i in range(1, 6):
    print(i)

#Controle de laços (break e continue)
#break → interrompe o laço.
#continue → pula a iteração atual e continua.

for i in range(1, 6):
    if i == 3:
        continue  # pula o número 3
    if i == 5:
        break     # para no 5
    print(i)

#Laços aninhados
Um laço dentro de outro.

for i in range(1, 4):
    for j in range(1, 4):
        print(i, j)
